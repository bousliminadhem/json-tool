let debounceTimer; // 1. IMPROVED HEALER: Handles structural gaps and missing braces function absoluteHealer(str) { let s = str.trim(); // Byte Scrub & Quote Normalization s = s.replace(/[\u00A0\u1680\u180e\u2000-\u200a\u202f\u205f\u3000\ufeff]/g, " "); s = s.replace(/'/g, '"'); // Key Quoting (Fixes unquoted keys) s = s.replace(/([{,]\s*)([a-zA-Z0-9._-]+)(\s*[:\s])/g, '$1"$2"$3'); // Missing Colons: Fixes "key" "value" patterns s = s.replace(/(")\s*(?!\s*:)\s*("|\d|-|true|false|null|{|\[)/g, '$1: $2'); // Missing Commas: Fixes "value" "key" patterns s = s.replace(/("|\d|true|false|null|}|\])\s+(")/g, '$1, $2'); // Structural wrap if (!s.startsWith('{') && !s.startsWith('[')) s = '{' + s + '}'; // Balance Braces: Crude but effective for truncated JSON const openBraces = (s.match(/{/g) || []).length; const closeBraces = (s.match(/}/g) || []).length; if (openBraces > closeBraces) { s += "}".repeat(openBraces - closeBraces); } // Clean up trailing commas s = s.replace(/,\s*([\]}])/g, '$1'); return s; } // 2. ORGANIZE: Alphabetical sorting of keys function sortObject(obj) { if (Array.isArray(obj)) return obj.map(sortObject); if (obj === null || typeof obj !== 'object') return obj; return Object.keys(obj).sort().reduce((acc, key) => { acc[key] = sortObject(obj[key]); return acc; }, {}); } function formatJSON(beautify = true) { const inputVal = document.getElementById('in').value; const out = document.getElementById('out'); const repairMsg = document.getElementById('repair-msg'); if(!inputVal.trim()) return; clearTimeout(debounceTimer); debounceTimer = setTimeout(() => { try { let processedData; let wasRepaired = false; try { // Try clean parse first processedData = JSON.parse(inputVal); } catch (e) { // Apply absolute repair const healed = absoluteHealer(inputVal); processedData = JSON.parse(healed); wasRepaired = true; } // Organize/Sort const organizedData = sortObject(processedData); repairMsg.style.display = wasRepaired ? "block" : "none"; render(organizedData, beautify); } catch (finalError) { out.innerHTML = `Unrepairable: ${finalError.message}`; } }, 150); } function render(data, beautify) { const out = document.getElementById('out'); const jsonString = JSON.stringify(data, null, beautify ? 4 : 0); // To show "updated parts", we'll highlight characters that didn't exist in original // This is a simplified "Additions" highlighter const original = document.getElementById('in').value; // Basic syntax highlighting for the output const highlighted = jsonString .replace(/&/g, '&').replace(//g, '>') .replace(/(".*?"|true|false|null|\d+)/g, (match) => { // If the match doesn't exist in the original text, highlight it as a repair return original.includes(match.replace(/"/g, '')) ? match : `${match}`; }); out.innerHTML = highlighted; }
